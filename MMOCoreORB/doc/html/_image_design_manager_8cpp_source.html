65 void ImageDesignManager::updateCustomization(CreatureObject* imageDesigner, const String& customizationName, float value, CreatureObject* creo) {
66  if (creo == NULL)
67  return;
68

91  if (customData->getIsScale()) {
92  float minScale = customData->getMinScale();
93  float maxScale = customData->getMaxScale();
94 
95  float height = minScale + value * (maxScale - minScale);
96 
97  creatureObject->setHeight(MAX(MIN(height, maxScale), minScale));
98 
99  return;
100  }
101 
102  Vector<String> fullVariables;
103  StringTokenizer tokenizer(variables);
104  tokenizer.setDelimeter(",");
105 
106  while (tokenizer.hasMoreTokens()) {
107  String var;
108  tokenizer.getStringToken(var);
109 
110  fullVariables.add(var);
111  }
112 
113  String appearanceFilename = creo->getObjectTemplate()->getAppearanceFilename();
114 
115  VectorMap<String, Reference<CustomizationVariable*> > variableLimits;
116 
117  AssetCustomizationManagerTemplate::instance()->getCustomizationVariables(appearanceFilename.hashCode(), variableLimits, false);
118 
119  for (int i = 0; i < fullVariables.size(); ++i) {
120  String var = fullVariables.get(i);
121 
122  for (int j = 0; j < variableLimits.size(); ++j) {
123  String fullVariableNameLimit = variableLimits.elementAt(j).getKey();
124 
125  if (fullVariableNameLimit.contains(var)) {
126  BasicRangedIntCustomizationVariable* ranged = dynamic_cast<BasicRangedIntCustomizationVariable*>(variableLimits.elementAt(j).getValue().get());
127 
128  if (ranged == NULL) {
129  error("variable " + fullVariableNameLimit + " is not ranged");
130 
131  continue;
132  }
133 
134  int min = ranged->getMinValueInclusive();
135  int max = ranged->getMaxValueExclusive();
136 
137  int count = max - min;
138 
139  int setVal;
140 
141  float currentValue = value;
142 
143  if (fullVariables.size() > 1) {
144  // examples for var count = 2
145  // ex: received value 0 is for i == 0 -> 1.0, i == 1 -> 0.0
146  // ex: received value 0.5 is for i == 0 -> 0.0, i == 1 -> 0.0
147  // ex: received value 1 is for i == 0 -> 0.0, i == 1 -> 1.0
148 
149  // pre: i Û [0, 1] && value Û [0, 1]
150  // post f Û [0, 1]
151  currentValue = MAX(0, ((value - 0.5) * 2) * (-1 + (i * 2)));
152  }
153 
154  if (customData->getReverse()) {
155  setVal = float(max - 1) - currentValue * (float(count) - 1);
156  } else {
157  setVal = float(min) + currentValue * (float(count) - 1);
158  }
159 
160  creatureObject->setCustomizationVariable(fullVariableNameLimit, setVal, true);
161 
162  //info("setting " + fullVariableNameLimit + " to " + String::valueOf(setVal), true);
163  }
164  }
165  }

300 void ImageDesignManager::loadCustomizationData() {
301  TemplateManager* templateManager = TemplateManager::instance();
302 
303  IffStream* iffStream = templateManager->openIffFile("datatables/customization/customization_data.iff");
304 
305  if (iffStream == NULL)
306  return;
307 
308  //Get the datatable, and parse it into a datatable object.
309  DataTableIff dataTable;
310  dataTable.readObject(iffStream);
311 
312  for (int i = 0; i < dataTable.getTotalRows(); i++) {
313  DataTableRow* dataRow = dataTable.getRow(i);
314 
315  if (dataRow == NULL)
316  continue;
317 
318  //Get the species gender
319  String speciesGender = dataRow->getCell(0)->toString();
320  uint32 templateCRC = String("object/creature/player/" + speciesGender + ".iff").hashCode();
321  PlayerCreatureTemplate* tmpl = dynamic_cast<PlayerCreatureTemplate*>(templateManager->getTemplate(templateCRC));
322 
323  if (tmpl == NULL)
324  continue;
325 
326  CustomizationDataMap* dataMap = tmpl->getCustomizationDataMap();
327 
328  if (dataMap == NULL)
329  continue;
330 
331  CustomizationData customizationData;
332  customizationData.parseRow(dataRow);
333 
334  customizationData.setMinScale(tmpl->getMinScale());
335  customizationData.setMaxScale(tmpl->getMaxScale());
336 
337  dataMap->put(customizationData.getCustomizationName(), customizationData);
338 
339  }
340 
341  //Done with the stream, so delete it.
342  if (iffStream != NULL) {
343  delete iffStream;
344  iffStream = NULL;
345  }

502 bool ImageDesignManager::validateCustomizationString(CustomizationVariables* data, const String& appearanceFilename, int skillLevel) {
503  VectorMap<String, Reference<CustomizationVariable*> > variables;
504  variables.setNullValue(NULL);
505  AssetCustomizationManagerTemplate::instance()->getCustomizationVariables(appearanceFilename.hashCode(), variables, false);
506 
507  if (variables.size() == 0) {
508  instance()->error("no customization data found for " + appearanceFilename);
509  return false;
510  }
511 
512  for (int i = 0; i < data->size(); ++i) {
513  uint8 id = data->elementAt(i).getKey();
514  int16 val = data->elementAt(i).getValue();
515 
516  String name = CustomizationIdManager::instance()->getCustomizationVariable(id);
517 
518  //instance()->info("validating " + name + " with value " + String::valueOf(val), true);
519 
520  CustomizationVariable* customizationVariable = variables.get(name).get();
521 
522  if (customizationVariable == NULL) {
523  instance()->error("customization variable id " + String::valueOf(id) + " not found in the appearance file " + appearanceFilename + " with value " + String::valueOf(val));
524 
525  continue;
526  }
527 
528  PaletteColorCustomizationVariable* palette = dynamic_cast<PaletteColorCustomizationVariable*>(customizationVariable);
529 
530  if (palette != NULL) {
531  if (!validatePalette(palette, val, skillLevel))
532  return false;
533  } else {
534  BasicRangedIntCustomizationVariable* range = dynamic_cast<BasicRangedIntCustomizationVariable*>(customizationVariable);
535 
536  if (range == NULL) {
537  instance()->error("unkown customization variable type " + name);
538  return false;
539  } else {
540  int maxExcl = range->getMaxValueExclusive();
541  int minIncl = range->getMinValueInclusive();
542 
543  if (val >= maxExcl || val < minIncl) {
544  instance()->error("variable outside bounds " + name + " value " + val + " outside bounds [" + String::valueOf(minIncl) + "," + String::valueOf(maxExcl) + ")");
545 
546  return false;
547  } else {
548  //instance()->info("variable " + name + " value " + String::valueOf(val) + " inside bounds [" + String::valueOf(minIncl) + "," + String::valueOf(maxExcl) + ")", true);
549  }
550 
551  }
552  }
553 
554 
555  //info("setting variable:" + name + " to " + String::valueOf(val), true);
556  }
557 
558  return true;
559 }
